**Tech Stack:** TypeScript, vitest, existing QwenLLM/Ollama wrappers, existing `cut()`/`burn()` from `task/video.ts`, existing `parseDanmu()` from `danmu/index.ts`.

```typescript
import type { DanmuItem, SC, Gift, AutoClipSignalConfig } from "@biliLive-tools/types";
import type { CandidateWindow, DanmuSample, SCSummary, DanmuStats, TimeWindow } from "./types";

// ====== 信号 A: 弹幕密度峰值 ======

export function detectDanmakuDensityPeaks(
  items: DanmuItem[],
  durationSec: number,
  config: AutoClipSignalConfig,
): TimeWindow[] {
  if (items.length === 0 || durationSec <= 0) return [];

  const buckets = buildBuckets(items, durationSec, config.bucketSec);
  const counts = buckets.map((b) => b.length);
  const mean = counts.reduce((a, b) => a + b, 0) / counts.length;
  const variance = counts.reduce((sum, c) => sum + (c - mean) ** 2, 0) / counts.length;
  const stdDev = Math.sqrt(variance);
  const threshold = mean + config.danmakuDensityThreshold * stdDev;

  // 标记热点桶
  const hotIndices: number[] = [];
  for (let i = 0; i < counts.length; i++) {
    if (counts[i] > threshold && counts[i] > 1) {
      hotIndices.push(i);
    }
  }

  // 合并相邻热点桶
  const merged = mergeAdjacentIndices(hotIndices, config);
  return merged.map(([startIdx, endIdx]) => {
    const startSec = Math.max(0, startIdx * config.bucketSec - config.windowPadding[0]);
    const endSec = Math.min(durationSec, endIdx * config.bucketSec + config.bucketSec + config.windowPadding[1]);
    return [startSec, endSec] as TimeWindow;
  });
}

function buildBuckets(items: DanmuItem[], durationSec: number, bucketSec: number): DanmuItem[][] {
  const bucketCount = Math.ceil(durationSec / bucketSec);
  const buckets: DanmuItem[][] = Array.from({ length: bucketCount }, () => []);
  for (const item of items) {
    const t = item.timestamp ?? (item.ts / 1000);
    const idx = Math.floor(t / bucketSec);
    if (idx >= 0 && idx < bucketCount) {
      buckets[idx].push(item);
    }
  }
  return buckets;
}

// ====== 信号 B: SC 爆发点 ======

export function detectSCBursts(sc: SC[], config: AutoClipSignalConfig): TimeWindow[] {
  if (sc.length === 0) return [];
  const results: TimeWindow[] = [];
  const windowSec = config.giftBurstWindowSec;

  // 滑动窗口扫描
  const sortedSc = [...sc].sort((a, b) => (a.timestamp ?? a.ts) - (b.timestamp ?? b.ts));

  for (let i = 0; i < sortedSc.length; i++) {
    const t0 = sortedSc[i].timestamp ?? (sortedSc[i].ts / 1000);
    let total = 0;
    let lastT = t0;
    for (let j = i; j < sortedSc.length; j++) {
      const tj = sortedSc[j].timestamp ?? (sortedSc[j].ts / 1000);
      if (tj - t0 > windowSec) break;
      total += Number(sortedSc[j].gift_price) || 0;
      lastT = tj;
    }
    if (total >= config.scMinAmount) {
      const center = (t0 + lastT) / 2;
      results.push([
        Math.max(0, center - config.windowPadding[0]),
        center + config.windowPadding[1],
      ]);
    }
  }
  return results;
}

// ====== 信号 C: 礼物潮 ======

export function detectGiftBursts(gifts: Gift[], config: AutoClipSignalConfig): TimeWindow[] {
  if (gifts.length === 0) return [];
  const results: TimeWindow[] = [];
  const windowSec = config.giftBurstWindowSec;

  const sorted = [...gifts].sort((a, b) => (a.timestamp ?? a.ts) - (b.timestamp ?? b.ts));

  for (let i = 0; i < sorted.length; i++) {
    const t0 = sorted[i].timestamp ?? (sorted[i].ts / 1000);
    let count = 0;
    let lastT = t0;
    for (let j = i; j < sorted.length; j++) {
      const tj = sorted[j].timestamp ?? (sorted[j].ts / 1000);
      if (tj - t0 > windowSec) break;
      count++;
      lastT = tj;
    }
    if (count >= config.giftBurstThreshold) {
      const center = (t0 + lastT) / 2;
      results.push([
        Math.max(0, center - config.windowPadding[0]),
        center + config.windowPadding[1],
      ]);
    }
  }
  return results;
}

// ====== 信号 D: 刷屏风暴（内容相似度） ======

export function detectBrushStorms(
  items: DanmuItem[],
  config: AutoClipSignalConfig,
): TimeWindow[] {
  if (items.length < 5) return [];

  const results: TimeWindow[] = [];
  const sorted = [...items].sort((a, b) => (a.timestamp ?? a.ts) - (b.timestamp ?? b.ts));

  // 滑动窗口, 检测短时间内的文本相似度
  const windowSec = 10; // 10s 滑动窗口

  for (let i = 0; i < sorted.length; i++) {
    const t0 = sorted[i].timestamp ?? (sorted[i].ts / 1000);
    const windowItems: DanmuItem[] = [];
    for (let j = i; j < sorted.length; j++) {
      const tj = sorted[j].timestamp ?? (sorted[j].ts / 1000);
      if (tj - t0 > windowSec) break;
      windowItems.push(sorted[j]);
    }

    if (windowItems.length < 5) continue;

    // 计算相似度矩阵 — 简单版: 取最长公共子串比例
    let similarPairs = 0;
    let totalPairs = 0;
    for (let a = 0; a < windowItems.length && totalPairs < 200; a++) {
      for (let b = a + 1; b < windowItems.length && totalPairs < 200; b++) {
        totalPairs++;
        const ta = windowItems[a].text ?? "";
        const tb = windowItems[b].text ?? "";
        if (ta.length < 2 || tb.length < 2) continue;
        const sim = longestCommonSubstringRatio(ta, tb);
        if (sim >= config.brushSimilarityThreshold) {
          similarPairs++;
        }
      }
    }

    if (totalPairs > 0 && similarPairs / totalPairs >= 0.3) {
      const tEnd = windowItems[windowItems.length - 1].timestamp
        ?? (windowItems[windowItems.length - 1].ts / 1000);
      results.push([
        Math.max(0, t0 - config.windowPadding[0]),
        tEnd + config.windowPadding[1],
      ]);
    }
  }
  return results;
}

/** 最长公共子串 / 较短文本长度 的比率 */
function longestCommonSubstringRatio(a: string, b: string): number {
  const shorter = a.length < b.length ? a : b;
  if (shorter.length === 0) return 0;

  let maxLen = 0;
  const dp = new Array(a.length + 1).fill(0).map(() => new Array(b.length + 1).fill(0));
  for (let i = 1; i <= a.length; i++) {
    for (let j = 1; j <= b.length; j++) {
      if (a[i - 1] === b[j - 1]) {
        dp[i][j] = dp[i-1][j-1] + 1;
        maxLen = Math.max(maxLen, dp[i][j]);
      }
    }
  }
  return maxLen / shorter.length;
}

// ====== 合并与去重 ======

export function mergeAndDeduplicate(
  windows: TimeWindow[],
  config: AutoClipSignalConfig,
): TimeWindow[] {
  if (windows.length === 0) return [];

  const sorted = [...windows].sort((a, b) => a[0] - b[0]);
  const merged: TimeWindow[] = [sorted[0]];

  for (let i = 1; i < sorted.length; i++) {
    const last = merged[merged.length - 1];
    if (sorted[i][0] <= last[1] + config.mergeGapSec) {
      last[1] = Math.max(last[1], sorted[i][1]);
    } else {
      merged.push([...sorted[i]]);
    }
  }

  // 裁剪: min < duration < max
  return merged
    .filter((w) => w[1] - w[0] >= config.minWindowDuration)
    .map((w) => {
      const dur = w[1] - w[0];
      if (dur > config.maxWindowDuration) {
        const mid = (w[0] + w[1]) / 2;
        return [mid - config.maxWindowDuration / 2, mid + config.maxWindowDuration / 2] as TimeWindow;
      }
      return w;
    });
}

function mergeAdjacentIndices(
  indices: number[],
  config: AutoClipSignalConfig,
): [number, number][] {
  if (indices.length === 0) return [];
  const maxGapBuckets = Math.ceil(config.mergeGapSec / config.bucketSec);

  const merged: [number, number][] = [[indices[0], indices[0]]];
  for (let i = 1; i < indices.length; i++) {
    const last = merged[merged.length - 1];
    if (indices[i] - last[1] <= maxGapBuckets) {
      last[1] = indices[i];
    } else {
      merged.push([indices[i], indices[i]]);
    }
  }
  return merged;
}

// ====== 主入口: 四路信号检测 + 合并 ======

export function detectSignals(
  stats: DanmuStats,
  config: AutoClipSignalConfig,
): CandidateWindow[] {
  const { danmu, sc, gift, duration } = stats;

  const windowsA = detectDanmakuDensityPeaks(danmu, duration, config).map(
    (w) => ({ timeRange: w, source: "density" as const }),
  );
  const windowsB = detectSCBursts(sc, config).map(
    (w) => ({ timeRange: w, source: "sc" as const }),
  );
  const windowsC = detectGiftBursts(gift, config).map(
    (w) => ({ timeRange: w, source: "gift" as const }),
  );
  const windowsD = detectBrushStorms(danmu, config).map(
    (w) => ({ timeRange: w, source: "brush" as const }),
  );

  // 合并相同信号源的相邻窗口 + 跨信号源去重
  const allWindows = [...windowsA, ...windowsB, ...windowsC, ...windowsD];
  const signalMap = new Map<string, Set<string>>();

  for (const { timeRange, source } of allWindows) {
    const key = `${timeRange[0].toFixed(1)}_${timeRange[1].toFixed(1)}`;
    if (!signalMap.has(key)) {
      signalMap.set(key, new Set());
    }
    signalMap.get(key)!.add(source);
  }

  const merged = mergeAndDeduplicate(
    [...signalMap.keys()].map((k) => {
      const [s, e] = k.split("_").map(Number);
      return [s, e] as TimeWindow;
    }),
    config,
  );

  // 构建 CandidateWindow
  return merged.map((timeRange) => {
    const signals = Array.from(signalMap.entries())
      .filter(([k]) => {
        const [s, e] = k.split("_").map(Number);
        return overlap([s, e], timeRange);
      })
      .flatMap(([, sources]) => Array.from(sources));

    const inWindow = danmu.filter((d) => {
      const t = d.timestamp ?? (d.ts / 1000);
      return t >= timeRange[0] && t <= timeRange[1];
    });
    const scInWindow = sc.filter((s) => {
      const t = s.timestamp ?? (s.ts / 1000);
      return t >= timeRange[0] && t <= timeRange[1];
    });
    const giftInWindow = gift.filter((g) => {
      const t = g.timestamp ?? (g.ts / 1000);
      return t >= timeRange[0] && t <= timeRange[1];
    });
    const uniqueUsers = new Set(inWindow.map((d) => d.user).filter(Boolean)).size;

    return {
      timeRange,
      signalSources: [...new Set(signals)],
      stats: {
        danmakuCount: inWindow.length,
        danmakuDensity: inWindow.length / (timeRange[1] - timeRange[0]),
        scTotal: scInWindow.reduce((sum, s) => sum + (Number(s.gift_price) || 0), 0),
        scCount: scInWindow.length,
        giftCount: giftInWindow.length,
        uniqueUsers,
        brushFrequency: 0, // 将在 pipeline 中填充
      },
      danmakuSample: inWindow.slice(0, 200).map((d) => ({
        timeOffset: (d.timestamp ?? (d.ts / 1000)) - timeRange[0],
        text: d.text ?? "",
        user: d.user,
      })),
      scSummary: scInWindow.map((s) => ({
        user: s.user ?? "",
        amount: Number(s.gift_price) || 0,
        message: s.text ?? "",
      })),
    };
  });
}

function overlap(a: TimeWindow, b: TimeWindow): boolean {
  return a[0] <= b[1] && b[0] <= a[1];
}
```

```typescript
import type { AutoClipLLMConfig } from "@biliLive-tools/types";
import type { CandidateWindow, ClipCandidateContext, LLMRankResult, HighlightSegment } from "./types";
import type { TimeWindow } from "./types";

const DEFAULT_PROMPT_TEMPLATE = `你是一个直播录像切片助手。以下是一段直播弹幕数据，请根据弹幕判断该时段是否值得切片。

弹幕统计: 共{count}条, 密度{density}条/秒, SC金额{scTotal}元,
         独立观众{uniqueUsers}人, 刷屏{brush}次

SC 记录:
{sc_records}

前后弹幕:
前: {before}
后: {after}

窗口内弹幕 (按时间排列):
{danmaku}

请返回 JSON（只返回 JSON，不要其他内容）:
{
  "isHighlight": true/false,
  "score": 0-10,
  "title": "切片标题(15字以内)",
  "tags": ["标签1", "标签2"],
  "highlightType": "funny/impressive/touching/hype/troll/not_highlight",
  "reason": "简短判断理由(20字以内)",
  "bestClipStart": 窗口内最佳起点秒数(相对于视频起点),
  "bestClipEnd": 窗口内最佳终点秒数(相对于视频起点)
}`;

export function buildLLMPrompt(ctx: ClipCandidateContext, template?: string): string {
  const t = template || DEFAULT_PROMPT_TEMPLATE;
  return t
    .replace("{count}", String(ctx.stats.danmakuCount))
    .replace("{density}", ctx.stats.danmakuDensity.toFixed(2))
    .replace("{scTotal}", String(ctx.stats.scTotal))
    .replace("{uniqueUsers}", String(ctx.stats.uniqueUsers))
    .replace("{brush}", String(ctx.stats.brushFrequency))
    .replace("{sc_records}", ctx.scSummary.length > 0
      ? ctx.scSummary.map((s) => `  ${s.user} ￥${s.amount}: ${s.message}`).join("\n")
      : "  无")
    .replace("{before}", ctx.surroundingBefore.length > 0
      ? ctx.surroundingBefore.slice(0, 10).join(" | ")
      : "无")
    .replace("{after}", ctx.surroundingAfter.length > 0
      ? ctx.surroundingAfter.slice(0, 10).join(" | ")
      : "无")
    .replace("{danmaku}", ctx.danmakuSamples.map((d, i) => `[${i}] ${d}`).join("\n"));
}

export function parseLLMResponse(raw: string, window: TimeWindow): LLMRankResult {
  try {
    // 提取 JSON 块（LLM 可能在 JSON 周围添加 markdown 或额外文本）
    const jsonMatch = raw.match(/\{[\s\S]*\}/);
    if (!jsonMatch) throw new Error("No JSON found in response");
    const obj = JSON.parse(jsonMatch[0]);

    const bestStart = clamp(
      Number(obj.bestClipStart) || window[0],
      window[0],
      window[1],
    );
    const bestEnd = clamp(
      Number(obj.bestClipEnd) || window[1],
      window[0],
      window[1],
    );

    return {
      isHighlight: Boolean(obj.isHighlight),
      score: Math.min(10, Math.max(0, Number(obj.score) || 0)),
      title: String(obj.title || "").slice(0, 15),
      tags: Array.isArray(obj.tags) ? obj.tags.map(String).slice(0, 5) : [],
      highlightType: validateHighlightType(obj.highlightType),
      reason: String(obj.reason || "").slice(0, 20),
      bestClipStart: bestStart,
      bestClipEnd: bestEnd,
    };
  } catch {
    return {
      isHighlight: false,
      score: 0,
      title: "",
      tags: [],
      highlightType: "not_highlight",
      reason: "LLM parse error",
      bestClipStart: window[0],
      bestClipEnd: window[1],
    };
  }
}

function clamp(v: number, min: number, max: number): number {
  return Math.max(min, Math.min(max, v));
}

function validateHighlightType(t: unknown): LLMRankResult["highlightType"] {
  const valid = ["funny", "impressive", "touching", "hype", "troll", "not_highlight"];
  return valid.includes(String(t)) ? String(t) as any : "not_highlight";
}

export function preRankCandidates(
  candidates: CandidateWindow[],
  maxCandidates: number,
): CandidateWindow[] {
  return [...candidates]
    .sort((a, b) => {
      const scoreA =
        a.stats.brushFrequency * 3 + a.stats.scTotal / 10 + a.stats.danmakuDensity;
      const scoreB =
        b.stats.brushFrequency * 3 + b.stats.scTotal / 10 + b.stats.danmakuDensity;
      return scoreB - scoreA;
    })
    .slice(0, maxCandidates);
}

/**
 * rankCandidates: 对候选窗口进行 LLM 精排
 * 
 * sendMessage 是实际 LLM 调用函数, 签名: (prompt: string) => Promise<string>
 */
export async function rankCandidates(
  candidates: CandidateWindow[],
  config: AutoClipLLMConfig,
  sendMessage: (prompt: string) => Promise<string>,
): Promise<HighlightSegment[]> {
  if (candidates.length === 0) return [];

  // 预截断 (如果候选过多)
  const toRank =
    candidates.length > config.maxCandidatesPerVideo
      ? preRankCandidates(candidates, config.maxCandidatesPerVideo)
      : candidates;

  // 构建调用列表
  const calls = toRank.map(async (candidate) => {
    const ctx = buildCandidateContext(candidate, config);
    const prompt = buildLLMPrompt(ctx, config.promptTemplate);
    const response = await sendMessage(prompt);
    const llmResult = parseLLMResponse(response, candidate.timeRange);

    return {
      timeRange: candidate.timeRange,
      bestRange: [llmResult.bestClipStart, llmResult.bestClipEnd] as TimeWindow,
      score: llmResult.score,
      title: llmResult.title,
      tags: llmResult.tags,
      highlightType: llmResult.highlightType,
      reason: llmResult.reason,
      signalSources: candidate.signalSources,
    } as HighlightSegment;
  });

  // 并行执行所有 LLM 调用
  const results = await Promise.all(calls);

  // 过滤非高光 + 按 score 排序 + 取 topK
  return results
    .filter((r) => r.score > 0)
    .sort((a, b) => b.score - a.score)
    .slice(0, config.topK);
}

function buildCandidateContext(
  candidate: CandidateWindow,
  config: AutoClipLLMConfig,
): ClipCandidateContext {
  const maxSamples = config.danmakuSampleMax;
  return {
    windowStart: candidate.timeRange[0],
    windowEnd: candidate.timeRange[1],
    danmakuSamples: candidate.danmakuSample.slice(0, maxSamples).map(
      (d) => `[${d.timeOffset.toFixed(0)}s] ${d.user ? `${d.user}: ` : ""}${d.text}`,
    ),
    scSummary: candidate.scSummary,
    stats: candidate.stats,
    surroundingBefore: [],
    surroundingAfter: [],
  };
}
```

```typescript
import path from "node:path";
import { v4 as uuidv4 } from "uuid";
import { parseDanmu } from "../danmu/index.js";
import { detectSignals } from "./signalDetector.js";
import { rankCandidates, preRankCandidates } from "./llmRanker.js";
import { cut } from "../task/video.js";
import logger from "../utils/log.js";

import type { AutoClipConfig, AutoClipPreset } from "@biliLive-tools/types";
import type { AutoClipResult } from "./types";
import type { DanmuStats } from "./types";

export type ProgressCallback = (
  stage: string,
  pct: number,
  message: string,
) => void;

export interface PipelineParams {
  videoPath: string;
  danmuPath: string;
  presetConfig: AutoClipConfig;
  onProgress?: ProgressCallback;
  sendMessage?: (prompt: string) => Promise<string>;
}

export async function runAutoClipPipeline(
  params: PipelineParams,
): Promise<AutoClipResult> {
  const { videoPath, danmuPath, presetConfig, onProgress, sendMessage } = params;
  const id = uuidv4();

  onProgress?.("parse", 0, "解析弹幕...");

  // 1. 解析弹幕
  const { danmu, sc, gift, guard, video_start_time } = await parseDanmu(danmuPath);
  const duration = await getVideoDuration(videoPath);

  const stats: DanmuStats = {
    danmu,
    sc,
    gift,
    guard,
    videoStartTime: video_start_time ?? 0,
    duration,
  };

  onProgress?.("detect", 20, `弹幕解析完成: ${danmu.length} 条弹幕, ${sc.length} 条SC`);

  // 2. Layer 1: 信号粗筛
  const candidates = detectSignals(stats, presetConfig.signal);

  if (candidates.length === 0) {
    onProgress?.("skip", 100, "未检测到高光信号, 跳过");
    return {
      id,
      videoPath,
      danmuPath,
      highlights: [],
      skipped: true,
      skippedReason: "no_signal",
    };
  }

  onProgress?.("detect", 50, `检测到 ${candidates.length} 个候选窗口`);

  // 3. Layer 2: LLM 精排
  let highlights;

  if (presetConfig.llm.enabled && sendMessage) {
    onProgress?.("rank", 60, "LLM 精排中...");
    highlights = await rankCandidates(candidates, presetConfig.llm, sendMessage);
    onProgress?.("rank", 80, `LLM 精排完成: 保留 ${highlights.length} 个高光片段`);
  } else {
    // 无 LLM 时用预排名代替
    const ranked = preRankCandidates(candidates, presetConfig.llm.topK);
    highlights = ranked.map((c) => ({
      timeRange: c.timeRange,
      bestRange: c.timeRange,
      score: 5,
      title: `高光片段`,
      tags: [],
      highlightType: "hype" as const,
      reason: "自动检测(无LLM)",
      signalSources: c.signalSources,
    }));
  }

  onProgress?.("done", 100, `完成: ${highlights.length} 个高光片段`);

  return {
    id,
    videoPath,
    danmuPath,
    highlights,
  };
}

async function getVideoDuration(videoPath: string): Promise<number> {
  try {
    const { readVideoMeta } = await import("../task/video.js");
    const meta = await readVideoMeta(videoPath);
    return meta?.format?.duration ?? 600;
  } catch {
    return 600; // 默认 10 分钟
  }
}

/** 触发实际切片导出 (调用现有 cut pipeline) */
export async function exportClips(
  videoPath: string,
  danmuPath: string,
  highlights: AutoClipResult["highlights"],
  config: AutoClipConfig["export"],
  onProgress?: ProgressCallback,
): Promise<string[]> {
  const outputFiles: string[] = [];

  for (let i = 0; i < highlights.length; i++) {
    const h = highlights[i];
    const outputName = config.namingTemplate
      .replace("{{title}}", h.title || "clip")
      .replace("{{index}}", String(i + 1))
      .replace("{{highlight_name}}", h.title || "");
    const outputPath = path.join(config.savePath, `${outputName}.${config.cutFormat}`);

    onProgress?.("cut", (i / highlights.length) * 100, `切片中: ${i + 1}/${highlights.length}`);

    try {
      await cut(
        [{ path: videoPath, ss: h.bestRange[0], to: h.bestRange[1] }],
        outputPath,
        config.ffmpegPresetId ? { presetId: config.ffmpegPresetId } : {},
        { disableMessage: true },
      );
      outputFiles.push(outputPath);
    } catch (error) {
      logger.error(`AutoClip export error for highlight ${i}`, error);
    }
  }

  return outputFiles;
}
```

```typescript
  };

  try {
    const result = await runAutoClipPipeline({
      videoPath,
      danmuPath,
      presetConfig,
      sendMessage,
      onProgress: (_stage, _pct, message) => {
        logger.info(`[AutoClip] ${message}`);
      },
    });

    resultCache.set(result.id, result);
    ctx.body = result;
  } catch (error: any) {
    logger.error("AutoClip run error:", error);
    ctx.status = 500;
    ctx.body = { error: error.message };
  }
});

router.get("/result/:id", async (ctx) => {
  const { id } = ctx.params;
  const result = resultCache.get(id);
  if (!result) {
    ctx.status = 404;
    ctx.body = { error: "Result not found" };
    return;
  }
  ctx.body = result;
});

function getDefaultConfig(): AutoClipConfig {
  const { AUTO_CLIP_DEFAULT_CONFIG } = require("@biliLive-tools/shared/presets/autoClipPreset");
  return AUTO_CLIP_DEFAULT_CONFIG;
}

export default router;
```
